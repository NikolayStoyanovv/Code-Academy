12. Напишете програма PrintPatterns, която отпечата следните текстови графики
на екрана. Графиките една под друга:
a)
#include <stdio.h>
int main() {
   short int rows=6, i, j, space;
   for (i = rows; i >= 1; --i) {
      for (space = 0; space < rows - i; ++space)
         printf("  ");
      for (j = i; j <= 2 * i - 1; ++j)
         printf("# ");
      for (j = 0; j < i - 1; ++j)
         printf("# ");
      printf("\n");
   }
   return 0;
}

b)
#include <stdio.h>
int main() {
   int i, space, rows=6, k = 0;
   for (i = 1; i <= rows; ++i, k = 0) {
      for (space = 1; space <= rows - i; ++space) {
         printf("  ");
      }
      while (k != 2 * i - 1) {
         printf("# ");
         ++k;
      }
      printf("\n");
   }
   return 0;
}
c)
#include <stdio.h>
int main() {
   short int rows=6, i, j, space,k=0;
   for (i = 1; i <= rows; ++i, k = 0) {
      for (space = 1; space <= rows - i; ++space) {
         printf("  ");
      }
      while (k != 2 * i - 1) {
         printf("# ");
         k++;
      }
      printf("\n");
   }
   for (i = rows-1; i >= 1; --i) {
      for (space = 0; space < rows - i; ++space)
         printf("  ");
      for (j = i; j <= 2 * i - 1; ++j)
         printf("# ");
      for (j = 0; j < i - 1; ++j)
         printf("# ");
      printf("\n");
   }
   return 0;
}
