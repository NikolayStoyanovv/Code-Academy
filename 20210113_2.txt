2. Да направим функция, която ще принтира числата от
десетичен в бинарен вид:
#include <stdio.h>
#include<math.h>
void bin(unsigned n) 
{ 
	unsigned i; 
	for (i = 1 << 7; i > 0; i = i >> 1 ) 
		(n & i)? printf("1"): printf("0"); 
	printf("\n");
}

int main(int argc, char *argv[]){
     char z=20;
    bin(z);
    
    return 0;}
