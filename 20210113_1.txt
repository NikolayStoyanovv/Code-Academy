#include <stdio.h>
#include<math.h>
void bin(unsigned n) 
{ 
	unsigned i; 
	for (i = 1 << 7; i > 0; i = i >> 1 ) 
		(n & i)? printf("1"): printf("0"); 
	printf("\n");
}

int main(int argc, char *argv[]){
    char z=255;
    bin(z);
    printf("%d\n",z);
    char a=10;
    
    printf("%d\n",a);
    bin(a);
    z=z+a;
    printf("%d\n",z);
    bin(z);
    return 0;}
отг:Ще се получи 9,защото прехвърляме диапазона на числото и то от 255 става 0.
