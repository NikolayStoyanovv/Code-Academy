7. Дефинирайте променливи със стойност 24 212, -1 357 674, 1 357 674, -1 357
674 000, 3 657 895 000.
Изведете всяка променлива на екрана със printf():
#include<stdio.h>
int main(){ 
short a=24212;
printf("%hd\n",a);//24212
int b=-1357674;
printf("%d\n",b);//-1357674
int c=1357674;
printf("%d\n",c);//1357674
int d=-1357674000;
printf("%d\n",d);//4294967292
unsigned  long int e=3 657 895 000;
printf("%lu",e);//3 657 895 000
return 0;
}
